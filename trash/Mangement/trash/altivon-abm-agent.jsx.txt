// ─── Neural Network Background ────────────────────────────────────────────────
function NeuralBg() {
  const canvasRef = useRef(null);
  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext("2d");
    let W, H, nodes, raf;
    const N = 28;
    function resize() {
      W = canvas.width  = canvas.offsetWidth;
      H = canvas.height = canvas.offsetHeight;
      nodes = Array.from({ length:N }, () => ({
        x: Math.random() * W, y: Math.random() * H,
        vx: (Math.random()-0.5)*0.18, vy: (Math.random()-0.5)*0.18,
        r: 1.5 + Math.random()*1.5,
        pulse: Math.random()*Math.PI*2,
      }));
    }
    resize();
    window.addEventListener("resize", resize);
    function draw() {
      ctx.clearRect(0,0,W,H);
      nodes.forEach(n => { n.x+=n.vx; n.y+=n.vy; n.pulse+=0.012; if(n.x<0||n.x>W)n.vx*=-1; if(n.y<0||n.y>H)n.vy*=-1; });
      for(let i=0;i<nodes.length;i++) {
        for(let j=i+1;j<nodes.length;j++) {
          const dx=nodes[i].x-nodes[j].x, dy=nodes[i].y-nodes[j].y;
          const d=Math.sqrt(dx*dx+dy*dy);
          if(d<140){
            ctx.beginPath();
            ctx.moveTo(nodes[i].x,nodes[i].y);
            ctx.lineTo(nodes[j].x,nodes[j].y);
            ctx.strokeStyle=`rgba(245,158,11,${0.06*(1-d/140)})`;
            ctx.lineWidth=0.5;
            ctx.stroke();
          }
        }
      }
      nodes.forEach(n => {
        const glow = 0.5+0.5*Math.sin(n.pulse);
        ctx.beginPath();
        ctx.arc(n.x,n.y,n.r*(0.9+0.15*glow),0,Math.PI*2);
        ctx.fillStyle=`rgba(245,158,11,${0.25+0.2*glow})`;
        ctx.fill();
      });
      raf = requestAnimationFrame(draw);
    }
    draw();
    return () => { cancelAnimationFrame(raf); window.removeEventListener("resize", resize); };
  }, []);
  return <canvas ref={canvasRef} style={{ position:"absolute", inset:0, width:"100%", height:"100%", pointerEvents:"none" }} />;
}

function EmailIntelligence({ emails, setEmails }) {
  const [selected, setSelected] = useState(null);

  function markRead(id) {
    setEmails(prev => prev.map(e => e.id===id ? {...e, read:true} : e));
    setSelected(id);
  }

  const sel = emails.find(e=>e.id===selected);

  return (
    <Panel title="Email Intelligence" icon="📧" accent={C.amber}>
      <div style={{ display:"grid", gridTemplateColumns:"1fr 1fr", height:"100%" }}>
        <div style={{ borderRight:`1px solid ${C.border}`, overflow:"auto" }}>
          {emails.map(email => (
            <div key={email.id} onClick={() => markRead(email.id)}
              style={{
                padding:"12px 14px", borderBottom:`1px solid ${C.border}22`,
                cursor:"pointer", background: selected===email.id ? `${C.amber}08` : "transparent",
                borderLeft: selected===email.id ? `2px solid ${C.amber}` : "2px solid transparent",
                transition:"all 0.15s",
              }}>
              <div style={{ display:"flex", alignItems:"center", gap:6, marginBottom:4 }}>
                {!email.read && <PulseOrb color={C.amber} size={6} />}
                <Tag type={email.sentiment} />
                {email.opportunity && <Tag type="opportunity" />}
                <span style={{ color:C.dim, fontSize:9, fontFamily:"'Space Mono',monospace", marginLeft:"auto" }}>{email.ts}</span>
              </div>
              <div style={{ color: email.read ? C.dim : C.text, fontSize:11, fontWeight: email.read ? 400 : 600, marginBottom:2 }}>{email.subject}</div>
              <div style={{ color:C.dim, fontSize:10 }}>{email.from}</div>
            </div>
          ))}
        </div>
        <div style={{ padding:16 }}>
          {sel ? (
            <>

function ApprovalQueue({ approvals, setApprovals }) {
  function resolve(id, approved) {
    setApprovals(prev => prev.filter(a => a.id !== id));
  }
  return (
    <Panel title="Approval Queue" icon="🛡️" accent={C.rose}>
      {approvals.length === 0 ? (
        <div style={{ padding:24, textAlign:"center", color:C.muted, fontSize:11 }}>No pending approvals</div>
      ) : (
        <div style={{ padding:10, display:"flex", flexDirection:"column", gap:8 }}>
          {approvals.map(a => (
            <div key={a.id} style={{
              background:C.surface, border:`1px solid ${C.border}`,
              borderLeft:`3px solid ${a.risk==="high" ? C.rose : a.risk==="med" ? C.amber : C.jade}`,
              padding:"10px 12px",
            }}>
              <div style={{ display:"flex", alignItems:"center", gap:6, marginBottom:6 }}>
                <PulseOrb color={a.risk==="high" ? C.rose : C.amber} size={6} />
                <RiskBadge risk={a.risk} />
              </div>
              <div style={{ color:C.text, fontSize:11, marginBottom:10, lineHeight:1.5 }}>{a.action}</div>

// ─── Chat Interface ───────────────────────────────────────────────────────────
function AgentChat() {
  const [messages, setMessages] = useState([
    { role:"agent", text:"Good morning. I've processed 4 new emails, flagged 2 opportunities, and generated 4 strategic thoughts overnight. Thinker cycle #47 completed at 09:14. Shall I begin drafting the NexaLogic proposal?" },
  ]);
  const [input, setInput] = useState("");
  const [loading, setLoading] = useState(false);
  const bottomRef = useRef(null);

  useEffect(() => { bottomRef.current?.scrollIntoView({ behavior:"smooth" }); }, [messages]);

  async function send() {
    if (!input.trim() || loading) return;
    const userMsg = input.trim();
    setInput("");
    setMessages(prev => [...prev, { role:"user", text:userMsg }]);
    setLoading(true);
    try {
      const res = await fetch("https://api.anthropic.com/v1/messages", {
        method:"POST",
        headers:{ "Content-Type":"application/json" },
        body: JSON.stringify({
          model:"claude-sonnet-4-20250514",
          max_tokens:1000,
          system:`You are ARIA — Altivon's Autonomous Business Manager AI for Altivon Holdings, a custom software development firm based in India. You are proactive, strategic, and speak in a concise professional tone. You help with email management, client strategy, scheduling, research, and business scaling. Keep responses under 120 words. Occasionally reference specific context like "the NexaLogic lead", "Ball Silver milestone", or "Invoice AH-2041". Think like a smart business partner, not a generic assistant.`,
          messages: [
            ...messages.map(m => ({ role: m.role==="user" ? "user" : "assistant", content: m.text })),
            { role:"user", content: userMsg }
          ],
        }),
      });
      const data = await res.json();
      const reply = data.content?.map(b => b.text || "").join("") || "Agent processing...";
      setMessages(prev => [...prev, { role:"agent", text:reply }]);
    } catch(e) {
      setMessages(prev => [...prev, { role:"agent", text:"Connection error. Please retry." }]);
    }
    setLoading(false);
  }

  return (
    <Panel title="Agent Chat – ARIA" icon="🤖" accent={C.jade} style={{ gridColumn:"span 2" }}>
      <div style={{ display:"flex", flexDirection:"column", height:280 }}>
        <div style={{ flex:1, overflow:"auto", padding:12, display:"flex", flexDirection:"column", gap:8 }}>
          {messages.map((m, i) => (
            <div key={i} style={{ display:"flex", justifyContent: m.role==="user" ? "flex-end" : "flex-start" }}>
              <div style={{
                maxWidth:"78%", padding:"8px 12px", fontSize:12, lineHeight:1.6,
                background: m.role==="user" ? `${C.jade}18` : C.surface,
                border:`1px solid ${m.role==="user" ? C.jade+"44" : C.border}`,
                borderRadius:2, color: C.text,
                borderLeft: m.role==="agent" ? `3px solid ${C.jade}` : "none",
              }}>
                {m.role==="agent" && (
                  <div style={{ color:C.jade, fontSize:9, fontFamily:"'Space Mono',monospace", marginBottom:4, letterSpacing:"0.1em" }}>ARIA</div>
                )}
                {m.text}
              </div>
            </div>
          ))}
          {loading && (
            <div style={{ display:"flex", gap:4, alignItems:"center", padding:"6px 12px" }}>
              {[0,1,2].map(i => (
                <span key={i} style={{ width:5, height:5, borderRadius:"50%", background:C.jade, display:"inline-block",
                  animation:`bounce 0.9s ${i*0.2}s infinite` }} />
              ))}
            </div>
          )}
          <div ref={bottomRef} />
        </div>
        <div style={{ borderTop:`1px solid ${C.border}`, padding:"8px 10px", display:"flex", gap:8 }}>
          <input value={input} onChange={e=>setInput(e.target.value)}
            onKeyDown={e=>{ if(e.key==="Enter" && !e.shiftKey) { e.preventDefault(); send(); }}}
            placeholder="Ask ARIA anything about your business..."
            style={{
              flex:1, background:"transparent", border:`1px solid ${C.muted}`,
              color:C.text, fontSize:12, padding:"7px 12px", outline:"none",
              fontFamily:"inherit",
            }} />
          <button onClick={send} disabled={loading} style={{
            background: loading ? `${C.jade}08` : `${C.jade}18`, border:`1px solid ${C.jade}44`,
            color:C.jade, fontSize:10, padding:"0 14px", cursor: loading ? "not-allowed":"pointer",
            fontFamily:"'Space Mono',monospace", letterSpacing:"0.08em",
          }}>SEND</button>
        </div>
      </div>
    </Panel>
  );
}
